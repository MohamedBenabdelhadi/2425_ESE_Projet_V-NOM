228void HAL_GPIO_EXTI_Callback(uint16_t GPIO_Pin)
229{
230 if (GPIO_Pin == AG_INT1_Pin) // Interrupt triggered by INT1
231 {
232 uint8_t int_source;
233 ADXL343_readRegister(ADXL343_INT_SOURCE, &int_source, 1);
234
235 if (int_source & 0x40) // Check SINGLE_TAP bit
236 {
237 DEBUG_PRINT("Single Tap Detected!\n");
238 Motor_Stop(&hMotors);
239 vTaskDelete(xControl);
240 }
241 }
242 else if (GPIO_Pin == AG_INT2_Pin) // Interrupt triggered by INT2
243 {
244 DEBUG_PRINT("Interrupt triggered on INT2\n");
245 // Handle other potential events from INT2 if needed
246 }
247 else if (GPIO_Pin == BTN_START_Pin) // Check if the interrupt is from BTN_START
248 {
249 static uint32_t last_interrupt_time = 0;
250 uint32_t current_time = HAL_GetTick();
251
252 // Debounce handling
253 if ((current_time - last_interrupt_time) > DEBOUNCE_DELAY_MS) {
254 buttonStartPressed++; // Set flag for button press
255 last_interrupt_time = current_time;
256 }
257 }
