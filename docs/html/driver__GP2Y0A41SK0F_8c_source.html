89void GP2Y0A41SK0F_get_distance(h_GP2Y0A41SK0F_t * htof)
90{
91 // Convert to Distance in units of mm
92 // by approximating datasheet graph
93 // using equation of form: y = a/x + b
94 // and two (x,y) points on the graph:
95 // (35mm, 3V) and (400mm, 0.3V)
96 const float a = 120;
97 const float b = 0.03;
98 float Vo = 0;
99
100 GP2Y0A41SK0F_Read_ToF1(htof);
101 GP2Y0A41SK0F_Read_ToF2(htof);
102
103 Vo = (GP2Y0A41SK0F_ADC_VCC * htof->adc_val_tof1) / GP2Y0A41SK0F_ADC_MAX_VALUE;
104
105 if ( Vo > b ) {
106 // Distance measured by ToF1 in mm
107 htof->distance_tof1 = (int)( a / (Vo - b) );
108 }
109
110 Vo = (GP2Y0A41SK0F_ADC_VCC * htof->adc_val_tof2) / GP2Y0A41SK0F_ADC_MAX_VALUE;
111
112 if ( Vo > b ) {
113 // Distance measured by ToF2 in mm
114 htof->distance_tof2 = (int)( a / (Vo - b) );
115 }
116}
