63void YLIDARX2_ProcessBuffer(h_YLIDARX2_t *lidar, uint16_t start, uint16_t end) {
64 for (uint16_t i = start; i < end - FRAME_LENGTH_MIN; ) {
65 uint16_t header = extract_uint16(&lidar->dmaBuffer[i]);
66
67 if (header == PACKET_HEADER) {
68 uint8_t ct = lidar->dmaBuffer[i + 2];
69 uint8_t lsn = lidar->dmaBuffer[i + 3];
70 uint16_t fsa = extract_uint16(&lidar->dmaBuffer[i + 4]);
71 uint16_t lsa = extract_uint16(&lidar->dmaBuffer[i + 6]);
72 uint16_t checksum = extract_uint16(&lidar->dmaBuffer[i + 8 + (lsn * 2)]);
73
74 uint16_t calculatedChecksum = 0;
75 for (uint16_t j = i; j < i + 8 + (lsn * 2); j++) {
76 calculatedChecksum ^= lidar->dmaBuffer[j];
77 }
78
79 if (calculatedChecksum == checksum) {
80 float startAngle = (fsa >> 1) / 64.0f;
81 float endAngle = (lsa >> 1) / 64.0f;
82 float angleIncrement = (endAngle - startAngle) / (lsn - 1);
83
84 for (uint8_t j = 0; j < lsn; j++) {
85 if (lidar->pointIndex >= YLIDARX2_MAX_POINTS) {
86 lidar->pointIndex = 0;
87 }
88
89 uint16_t distance = extract_uint16(&lidar->dmaBuffer[i + 8 + (j * 2)]);
90 float angle = startAngle + (j * angleIncrement) - 180;
91 lidar->points[lidar->pointIndex].angle = angle;
92 lidar->points[lidar->pointIndex].distance = distance / 4.0f;
93 lidar->points[lidar->pointIndex].intensity = (distance & 0x01) ? 1 : 0;
94
95 DEBUG_PRINT("Point %d: Angle %.2f°, Distance %.2f mm, Intensity %u\r\n",
96 lidar->pointIndex,
97 lidar->points[lidar->pointIndex].angle,
98 lidar->points[lidar->pointIndex].distance,
99 lidar->points[lidar->pointIndex].intensity);
100
101 lidar->pointIndex++;
102 }
103
104 i += 10 + (lsn * 2); // Move to next packet
105 } else {
106 DEBUG_PRINT("Checksum failed at index %d\r\n", i);
107 i++;
108 }
109 } else {
110 i++;
111 }
112 }
