120void Motor_UpdateSpeed(h_Motor_t * hMotors) {
121 // Increase speed progressively
122 if (hMotors->speed1 > hMotors->current_speed1)
123 hMotors->current_speed1 += hMotors->speed_increase_rate1;
124 if (hMotors->speed2 > hMotors->current_speed2)
125 hMotors->current_speed2 += hMotors->speed_increase_rate2;
126
127 // Decrease speed progressively
128 if (hMotors->speed1 < hMotors->current_speed1)
129 hMotors->current_speed1 -= hMotors->speed_increase_rate1;
130 if (hMotors->speed2 < hMotors->current_speed2)
131 hMotors->current_speed2 -= hMotors->speed_increase_rate2;
132
133 __HAL_TIM_SET_COMPARE(hMotors->htim, TIM_CHANNEL_1, hMotors->current_speed1);
134 __HAL_TIM_SET_COMPARE(hMotors->htim, TIM_CHANNEL_2, hMotors->current_speed2);
135}
