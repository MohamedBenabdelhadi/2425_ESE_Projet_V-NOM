161void ADXL343_ConfigureTap(h_ADXL343_t *hadxl)
162{
163 uint8_t val;
164
165 // Set tap threshold
166 val = 50; // 50 * 0.0625g per LSB
167 ADXL343_writeRegister(ADXL343_THRESH_TAP, &val, 1);
168
169 // Set tap duration
170 val = 30; // 30 ms
171 ADXL343_writeRegister(ADXL343_DUR, &val, 1);
172
173 // Enable tap detection on all axes
174 val = 0x07; // X, Y, Z enabled
175 ADXL343_writeRegister(ADXL343_TAP_AXES, &val, 1);
176
177 // Enable single-tap interrupt
178 val = 0x40; // SINGLE_TAP bit
179 ADXL343_writeRegister(ADXL343_INT_ENABLE, &val, 1);
180
181 // Map single-tap interrupt to INT1
182 val = 0x00; // Map to INT1
183 ADXL343_writeRegister(ADXL343_INT_MAP, &val, 1);
184
185 // Configure GPIO for INT1
186 GPIO_InitTypeDef GPIO_InitStruct = {0};
187 __HAL_RCC_GPIOA_CLK_ENABLE();
188 GPIO_InitStruct.Pin = AG_INT1_Pin;
189 GPIO_InitStruct.Mode = GPIO_MODE_IT_RISING;
190 GPIO_InitStruct.Pull = GPIO_NOPULL;
191 HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);
192
193 // Enable EXTI interrupt
194 HAL_NVIC_SetPriority(EXTI3_IRQn, 5, 0);
195 HAL_NVIC_EnableIRQ(EXTI3_IRQn);
196
197 DEBUG_PRINT("ADXL343 configured for single-tap detection.\n");
198}

206void ADXL343_get_Acceleration(h_ADXL343_t *hadxl)
